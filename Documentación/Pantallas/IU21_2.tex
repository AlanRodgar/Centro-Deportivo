\subsection{IU21.2 Pantalla de alta de curso en el sistema}

\subsubsection{Objetivo}
	El gerente de Sucursal tiene la facultad de dar de alta de Curso,

\subsubsection{Diseño}
	Esta pantalla aparece al verificar las credenciales con su correspindiente nombre de usuario y contraseña del gerente de sucursal. 

\IUfig[.5]{gui/IU21-2}{IU21.2}{Pantalla de alta de curso en el sistema.}

\subsubsection{Salidas}

	Ninguna.

\subsubsection{Entradas}
nombre de curso, costo, horario, fecha de inicio,area, fecha de termino,nombre de instructor y su  horario.

\subsubsection{Comandos}
\begin{itemize}
	\item \IUbutton{Alta de Curso}:Acceso a la opcion de registrar curso en el sistema. Al seleccionar esta pestaña, se muestra la \IUref{IU21.2}{Pantalla de Alta de curso}.

\end{itemize}

\subsubsection{Mensajes}
	\begin{Citemize}
		\item {\bf MSG5} Error al cargar pantalla \IUref{IU21.2}{Pantalla de Alta de curso}.
	\end{Citemize}
